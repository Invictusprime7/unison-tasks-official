  <script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"
     integrity="sha256-200KjU9F/p4NxAoCcT8fA2kIq1dG8yHj9F/8I9p0r2C/oD+A3C08="

  <script>
    (function() {
      'use strict';

      // --- Mobile Menu functionality ---
      const mobileMenuButton = document.getElementById('mobile-menu-button');
      const mobileMenu = document.getElementById('mobile-menu');

      if (mobileMenuButton && mobileMenu) {
        mobileMenuButton.addEventListener('click', () => {
          mobileMenu.classList.toggle('hidden');
          mobileMenuButton.setAttribute('aria-expanded', mobileMenu.classList.contains('hidden') ? 'false' : 'true');
        });

        mobileMenu.querySelectorAll('a').forEach(link => {
          link.addEventListener('click', () => {
            mobileMenu.classList.add('hidden');
            mobileMenuButton.setAttribute('aria-expanded', 'false');
          });
        });
      }

      // --- Smooth scrolling for navigation links ---
      document.querySelectorAll('a[href^="#"]').forEach(anchor => {
        anchor.addEventListener('click', function (e) {
          e.preventDefault();

          const targetId = this.getAttribute('href');
          const targetElement = document.querySelector(targetId);

          if (targetElement) {
            targetElement.scrollIntoView({
              behavior: 'smooth'
            });
          }
        });
      });

      // --- Appointment Booking Calendar Logic ---
      const currentMonthYearDisplay = document.getElementById('currentMonthYear');
      const calendarDaysContainer = document.getElementById('calendarDays');
      const prevMonthBtn = document.getElementById('prevMonth');
      const nextMonthBtn = document.getElementById('nextMonth');
      const selectedDateDisplay = document.getElementById('selectedDateDisplay');
      const bookingForm = document.getElementById('bookingForm');
      const bookingMessage = document.getElementById('bookingMessage');

      let currentCalendarDate = new Date(); // Represents the month/year currently displayed
      let selectedDate = null; // Stores the user selected Date object

      function renderCalendar() {
        calendarDaysContainer.innerHTML = ''; // Clear previous days
        const year = currentCalendarDate.getFullYear();
        const month = currentCalendarDate.getMonth(); // 0-indexed

        currentMonthYearDisplay.textContent = new Date(year, month).toLocaleString('en-US', { month: 'long', year: 'numeric' });

        const firstDayOfMonth = new Date(year, month, 1).getDay();
        const daysInMonth = new Date(year, month + 1, 0).getDate();
        const daysInPrevMonth = new Date(year, month, 0).getDate();

        const today = new Date();
        today.setHours(0, 0, 0, 0);

        for (let i = firstDayOfMonth; i > 0; i--) {
          const day = document.createElement('div');
          day.className = 'calendar-day disabled';
          day.innerHTML = `<span class="calendar-day-number">${daysInPrevMonth - i + 1}</span>`;
          calendarDaysContainer.appendChild(day);
        }

        for (let i = 1; i <= daysInMonth; i++) {
          const day = document.createElement('div');
          const date = new Date(year, month, i);
          date.setHours(0, 0, 0, 0);

          day.classList.add('calendar-day', 'current-month');
          day.innerHTML = `<span class="calendar-day-number">${i}</span>`;
          day.dataset.date = date.toISOString();

          if (date < today) {
            day.classList.add('disabled');
            day.style.cursor = 'not-allowed';
          } else {
            day.addEventListener('click', () => selectDate(date, day));
          }

          if (date.getTime() === today.getTime()) {
            day.classList.add('today');
          }

          if (selectedDate && date.getTime() === selectedDate.getTime()) {
            day.classList.add('selected');
          }

          calendarDaysContainer.appendChild(day);
        }

        const totalDaysRendered = firstDayOfMonth + daysInMonth;
        const remainingCells = 42 - totalDaysRendered;

        for (let i = 1; i <= remainingCells; i++) {
          const day = document.createElement('div');
          day.className = 'calendar-day disabled';
          day.innerHTML = `<span class="calendar-day-number">${i}</span>`;
          calendarDaysContainer.appendChild(day);
        }
      }

      function selectDate(date, element) {
        const previouslySelected = document.querySelector('.calendar-day.selected');
        if (previouslySelected) {
          previouslySelected.classList.remove('selected');
        }

        element.classList.add('selected');
        selectedDate = date;

        selectedDateDisplay.querySelector('span').textContent = selectedDate.toLocaleDateString('en-US', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });
        selectedDateDisplay.classList.remove('hidden');
      }

      function navigateMonth(direction) {
        currentCalendarDate.setMonth(currentCalendarDate.getMonth() + direction);
        renderCalendar();
      }

      prevMonthBtn.addEventListener('click', () => navigateMonth(-1));
      nextMonthBtn.addEventListener('click', () => navigateMonth(1));

      renderCalendar();

      bookingForm.addEventListener('submit', function(event) {
        event.preventDefault();

        bookingMessage.classList.add('hidden');
        bookingMessage.classList.remove('text-green-600', 'text-red-600');

        if (!selectedDate) {
          bookingMessage.textContent = 'Please select a date from the calendar.';
          bookingMessage.classList.add('text-red-600');
          bookingMessage.classList.remove('hidden');
          return;
        }

        const service = document.getElementById('service').value;
        const name = document.getElementById('name').value;
        const email = document.getElementById('email').value;
        const notes = document.getElementById('notes').value;

        if (!service || !name || !email) {
          bookingMessage.textContent = 'Please fill in all required fields (Service, Name, Email).';
          bookingMessage.classList.add('text-red-600');
          bookingMessage.classList.remove('hidden');
          return;
        }

        console.log('--- Booking Submitted ---');
        console.log('Service:', service);
        console.log('Name:', name);
        console.log('Email:', email);
        console.log('Selected Date:', selectedDate.toLocaleDateString());
        console.log('Notes:', notes);

        setTimeout(() => {
          bookingMessage.textContent = `Booking for ${service} on ${selectedDate.toLocaleDateString()} confirmed, ${name}! Check your email.`;
          bookingMessage.classList.add('text-green-600');
          bookingMessage.classList.remove('hidden');
          bookingForm.reset();
          const previouslySelected = document.querySelector('.calendar-day.selected');
          if (previouslySelected) {
            previouslySelected.classList.remove('selected');
          }
          selectedDate = null;
          selectedDateDisplay.classList.add('hidden');
        }, 1000);
      });


      // --- Sign Up Modal Logic ---
      const signupModal = document.getElementById('signupModal');
      const closeSignupModalBtn = document.getElementById('closeSignupModal');
      const openSignupModalBtns = document.querySelectorAll('.js-open-signup-modal');
      const signupForm = document.getElementById('signupForm');
      const signupMessage = document.getElementById('signupMessage');
      const loginLink = document.querySelector('.js-login-link');

      function showSignupModal() {
        signupModal.classList.add('active');
        document.body.style.overflow = 'hidden';
      }

      function hideSignupModal() {
        signupModal.classList.remove('active');
        document.body.style.overflow = '';
        signupForm.reset();
        signupMessage.classList.add('hidden');
        signupMessage.classList.remove('text-green-600', 'text-red-600');
      }

      openSignupModalBtns.forEach(button => {
        button.addEventListener('click', showSignupModal);
      });

      closeSignupModalBtn.addEventListener('click', hideSignupModal);
      signupModal.addEventListener('click', (e) => {
        if (e.target === signupModal) {
          hideSignupModal();
        }
      });
      document.addEventListener('keydown', (e) => {
        if (e.key === 'Escape' && signupModal.classList.contains('active')) {
          hideSignupModal();
        }
      });

      signupForm.addEventListener('submit', function(event) {
        event.preventDefault();

        signupMessage.classList.add('hidden');
        signupMessage.classList.remove('text-green-600', 'text-red-600');

        const email = document.getElementById('signupEmail').value;
        const password = document.getElementById('signupPassword').value;
        const confirmPassword = document.getElementById('confirmPassword').value;

        if (password !== confirmPassword) {
          signupMessage.textContent = 'Passwords do not match!';
          signupMessage.classList.add('text-red-600');
          signupMessage.classList.remove('hidden');
          return;
        }

        if (password.length < 6) {
            signupMessage.textContent = 'Password must be at least 6 characters long.';
            signupMessage.classList.add('text-red-600');
            signupMessage.classList.remove('hidden');
            return;
        }

        console.log('--- Sign Up Attempt ---');
        console.log('Email:', email);
        console.log('Password (hashed in real app!):', password);

        setTimeout(() => {
          const success = Math.random() > 0.2;

          if (success) {
            signupMessage.textContent = `Account created successfully for ${email}! You can now log in.`;
            signupMessage.classList.add('text-green-600');
            signupMessage.classList.remove('hidden');
            signupForm.reset();
          } else {
            signupMessage.textContent = 'Sign up failed. Please try again or use a different email.';
            signupMessage.classList.add('text-red-600');
            signupMessage.classList.remove('hidden');
          }
        }, 1500);
      });

      loginLink.addEventListener('click', function(e) {
        e.preventDefault();
        console.log('Redirecting to login / showing login form...');
        alert('This would switch to a login form or page in a real application!');
        hideSignupModal();
      });

      // --- Bubble Animation Logic ---
      const bubblesContainer = document.querySelector('.bubbles-container');
      const numberOfBubbles = 25;

      function createBubble() {
        const bubble = document.createElement('div');
        bubble.classList.add('bubble');

        const size = Math.random() * 60 + 20;
        bubble.style.width = `${size}px`;
        bubble.style.height = `${size}px`;

        bubble.style.left = `${Math.random() * 100}%`;

        bubble.style.animationDuration = `${Math.random() * 10 + 10}s`;
        bubble.style.animationDelay = `${Math.random() * numberOfBubbles * 0.5}s`;

        bubblesContainer.appendChild(bubble);

        bubble.addEventListener('animationend', () => {
          bubble.remove();
          createBubble();
        });
      }

      for (let i = 0; i < numberOfBubbles; i++) {
        createBubble();
      }

      // --- Leaflet Map Logic ---
      // Ensure the map container exists before initializing
      const mapElement = document.getElementById('map');
      if (mapElement) {
          const map = L.map('map').setView([0, 0], 2); // Centered on [latitude, longitude] with zoom level 2

          L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
              attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors'
          }).addTo(map);

          // Add a sample marker
          const sampleMarker = L.marker([51.5, -0.09]).addTo(map) // London coordinates
              .bindPopup('<b>Hello world!</b><br />We have places to book here.')
              .openPopup();

          map.on('click', function(e) {
              L.popup()
                  .setLatLng(e.latlng)
                  .setContent(`You clicked the map at ${e.latlng.toString()}`)
                  .openOn(map);
          });
      }

    })();
  </script>
